### undefined-error
An exception class to throw undefined error.

**Example**  
```js
const {UndefinedError} = require('@9fv.io/undefined-error')

function test(name) {
  if (!name) {
    throw new UndefinedError('an error') // throw an error of type {UndefinedError}.
  }
  return name;
}
```
